import React from 'react';
import {
    View,
    Text,
    Button,
    StyleSheet
} from 'react-native';
import {
    useTheme
} from '../styles/theme'; // Import the useTheme hook

const ErrorMessage = ({
    message,
    onRetry
}) => {
    const {
        theme
    } = useTheme(); // Access the theme

    return ( <
        View style = {
            [styles.container, {
                backgroundColor: theme.background
            }]
        } >
        <
        Text style = {
            [styles.text, {
                color: theme.error
            }]
        } > Error: {
            message
        } < /Text> {
            onRetry && < Button title = "Retry"
            onPress = {
                onRetry
            }
            color = {
                theme.primary
            }
            />} <
            /View>
        );
    };

    const styles = StyleSheet.create({
        container: {
            flex: 1,
            justifyContent: 'center',
            alignItems: 'center',
            padding: 20,
        },
        text: {
            fontSize: 16,
            marginBottom: 10,
            textAlign: 'center',
        },
    });

    export default ErrorMessage;
